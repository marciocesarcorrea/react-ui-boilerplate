# GlobalSnackbar
Call snackbar from anywhere, it's very simple!

## Code Example
```jsx
import React from "react";

import { Button } from "@material-ui/core"

import { snack, showSnack } from "components/GlobalSnackbar";

const MyComponent: React.FC = () => {
  const customSnackHandleClick = () => {
    showSnack("Custom Snack", {
      action: (key) => <Button onClick={() => alert('Clicked')}>Click Here!</Button>
    });
  }

  return (
    <Button onClick={() => snack.info("Snack Info")}>Info</Button>
    <Button onClick={() => snack.warning("Snack Warning")}>Warning</Button>
    <Button onClick={() => snack.success("Snack Success") }>Success</Button>
    <Button onClick={() => snack.error("Snack Error")}>Error</Button>
    <Button onClick={customSnackHandleClick}>Custom snack</Button>
  )
}

```
